# cub3d
## 1. 실행
~~~bash
#실행
git clone https://github.com/tapark/cub3d.git && cd cub3d && make && ./cub3D maps/map2.cub
#종료
press_key[ESC] or click[X]
#스크린샷
./cub3D maps/map2.cub --save
#초기화
make fclean
~~~
